5

8the same global batch -- only the parallelism layout differs:
9

68def _param_count(cfg: GrugModelConfig) -> tuple[float, float]:
69    """(total, active) parameter counts for the production grug-MoE config, in billions.
70 
71    Per layer: attention ~4*D^2 (square q/k/v/o), the MoE block has 3 expert matrices
72    of [E, D, I] (gate/up/down) plus a [D, E] router, and a shared expert of 3
73    [D, I_shared] matrices. "Active" replaces all-experts with the top-k routed
74    experts (6 * ept * D * I), keeping the shared expert and router. Embed + untied
75    head are 2 * V * D.
76    """
77    d = cfg.hidden_dim
78    e = cfg.num_experts
79    i = cfg.intermediate_dim
80    ept = cfg.num_experts_per_token
81    i_shared = cfg.shared_expert_intermediate_dim
82    attn = 4 * d * d
83    router = d * e
84    shared = 3 * d * i_shared
85    moe_total = 6 * e * d * i + router + shared
86    moe_active = 6 * ept * d * i + router + shared
87    embed_head = 2 * cfg.vocab_size * d
88    total = embed_head + cfg.num_layers * (attn + moe_total)
89    active = embed_head + cfg.num_layers * (attn + moe_active)

93def _timed_steps(step_fn, state, data_fn, *, warmup: int, iters: int) -> float:
94    """Return mean seconds/step over ``iters`` steps after ``warmup`` warmup steps."""
95    for i in range(warmup):
96        state = step_fn(state, data_fn(i))
97    jax.block_until_ready(state)
98    start = time.perf_counter()
99    for i in range(iters):
100        state = step_fn(state, data_fn(warmup + i))
101    jax.block_until_ready(state)
102    return (time.perf_counter() - start) / iters

289        # production `next_token_loss` (sum over layers / num_layers).
290        ce_total = jax.lax.psum(jnp.where(is_last, ce, 0.0), STAGE_AXIS)

296    # input sharding to match in_specs exactly).

304    loss_weight = _place(loss_weight, repl)

307    # the body, which the production model requires -- it reshards activations with

309    # default check_vma=True those axes become Auto inside the shard_map and the

313    return shard_map(
314        body,
315        mesh=mesh,
316        in_specs=(stage_in_specs, embed_in_specs, stage_spec, repl, repl),
317        out_specs=repl,
318        axis_names=frozenset({STAGE_AXIS}),
319        check_vma=False,
320    )(stage_block_arrays, embed_arrays, layer_masks, token_ids, loss_weight)

383def pipeline_value_and_grad(

392    num_microbatches: int,
393) -> tuple[jax.Array, eqx.Module, eqx.Module]:
394    """``(loss, embed_grads, stage_grads)`` for the production Transformer via a
395    manual GPipe backward inside the stage-manual ``shard_map``.
396 
397    The whole model -- embed (stage 0), the stacked production ``Block``s (per
398    stage), and the fused-CE head (last stage) -- is differentiated without ever
399    transposing the outer shard_map: a forward sweep captures per-(stage,

401    on the last stage and ``ppermute``s activation-cotangents stage->stage-1.

404    microbatch, seq]``. Returns ``(loss, embed_grads, stage_grads)`` where
405    ``embed_grads`` is the array tree of the replicated embed/norm/head params (the
406    cotangent psummed onto every stage; only stages 0 / last contribute) and
407    ``stage_grads`` is the ``[stage, layers_per_stage, ...]`` block grad (each
408    stage keeps its own shard).
409    """
410    num_layers = len(transformer.blocks)
411    cfg = transformer.config
412    seq_len = token_microbatches.shape[-1]
413 
414    embed_arrays, embed_static = eqx.partition(
415        (
416            transformer.token_embed,
417            transformer.embed_norm,
418            transformer.embed_gated_norm,
419            transformer.final_norm,
420            transformer.final_gated_norm,
421            transformer.output_proj,
422        ),
423        eqx.is_array,
424    )
425 
426    layer_masks = build_layer_masks(transformer, num_stages, seq_len)
427 
428    stage_spec = P(STAGE_AXIS)
429    repl = P()
430    embed_in_specs = jax.tree_util.tree_map(lambda _: repl, embed_arrays)
431    stage_in_specs = jax.tree_util.tree_map(lambda _: stage_spec, stage_block_arrays)
432 
433    def body(stage_arrays, embed, masks, tokens, weights):
434        sid = jax.lax.axis_index(STAGE_AXIS)
435        S = num_stages
436        M = num_microbatches
437        T = M + S - 1
438        fwd_perm = [(i, i + 1) for i in range(S - 1)]
439        bwd_perm = [(i + 1, i) for i in range(S - 1)]
440        is_first = sid == 0
441        is_last = sid == (S - 1)
442        batch_spec = _batch_spec()
443 
444        token_embed, embed_norm, embed_gated_norm, final_norm, final_gated_norm, output_proj = eqx.combine(
445            embed, embed_static
446        )
447        stage_blocks = jax.tree_util.tree_map(lambda x: x[0], stage_arrays)
448        stage_masks = masks[0]
449 
450        hidden_shape = (token_microbatches.shape[1], seq_len, cfg.hidden_dim)
451        # Every activation buffer (ppermuted carry, last-stage collection, backward
452        # cotangent carry) lives in the Explicit batch sharding the production blocks
453        # reshard to, so the wavefront selects / dynamic-updates all agree.
454        zero_hidden = reshard(jnp.zeros(hidden_shape, jnp.float32), batch_spec)
455 
456        # --- Forward sweep: capture per-timestep vjp closures + last-stage hiddens.
457        vjp_x_by_t: list = [None] * T
458        vjp_p_by_t: list = [None] * T
459        embed_vjp_by_t: list = [None] * T
460        valid_by_t: list = [None] * T
461 
462        buf = zero_hidden
463        z_total = jnp.zeros((), jnp.float32)
464        h_final = jnp.broadcast_to(zero_hidden, (M, *hidden_shape))
465        for t in range(T):
466            m = t - sid
467            valid = (m >= 0) & (m < M)
468            m_clip = jnp.clip(m, 0, M - 1)
469            tok_m = jax.lax.dynamic_index_in_dim(tokens, m_clip, axis=0, keepdims=False)
470 
471            # Loop vars bound as lambda defaults so jax.vjp captures this iteration's
472            # tensors immediately (B023-safe).
473            embedded, embed_vjp = jax.vjp(
474                lambda te, en, eg, _t=tok_m: _embed_tokens(te, en, eg, _t, batch_spec),
475                token_embed,
476                embed_norm,
477                embed_gated_norm,
478            )
479            # Both select branches must share a sharding: the embed gather carries the
480            # Explicit batch spec, buf (ppermuted activation) starts replicated, so
481            # reshard both to the batch spec before choosing.
482            stage_in = jnp.where(is_first, reshard(embedded, batch_spec), reshard(buf, batch_spec))
483 
484            (stage_out, z_local), vjp_x = jax.vjp(
485                lambda h, _sb=stage_blocks: _run_stage_blocks(_sb, block_static, h, stage_masks), stage_in
486            )
487            (_, _), vjp_p = jax.vjp(
488                lambda sb, _si=stage_in: _run_stage_blocks(sb, block_static, _si, stage_masks), stage_blocks
489            )
490 
491            z_total = z_total + jnp.where(valid, z_local, 0.0)
492            contrib = jnp.where(is_last & valid, stage_out, jnp.zeros_like(stage_out))
493            prev = jax.lax.dynamic_index_in_dim(h_final, m_clip, axis=0, keepdims=False)
494            h_final = jax.lax.dynamic_update_index_in_dim(h_final, prev + contrib, m_clip, axis=0)
495 
496            vjp_x_by_t[t] = vjp_x
497            vjp_p_by_t[t] = vjp_p
498            embed_vjp_by_t[t] = embed_vjp
499            valid_by_t[t] = valid
500 
501            buf = jax.lax.ppermute(stage_out, STAGE_AXIS, fwd_perm)
502 
503        # --- Head: scored once after the sweep on the last stage's collected hiddens.
504        # h_final is psummed so the last stage's per-microbatch hidden is visible on
505        # every stage (every other stage held zero), making the head vjp run identically
506        # on all stages -- so its (replicated) grad is already correct without a psum.
507        h_final = jax.lax.psum(h_final, STAGE_AXIS)
508        # Use the in-body (stage-Manual mesh) token/weight args -- the outer closure
509        # vars carry the all-Explicit mesh aval and a reshape under them clashes.
510        flat_batch = M * hidden_shape[0]
511        labels = _next_token_labels(tokens.reshape(flat_batch, seq_len))
512        weights_flat = weights.reshape(flat_batch, seq_len)
513        # Merging the replicated microbatch axis M into the batch-sharded microbatch
514        # axis is ambiguous to the partitioner, so name the flattened batch's
515        # sharding explicitly (the merged [M*microbatch] axis carries the batch spec).
516        flat_hidden_spec = P(batch_spec[0], None, None)
517        h_final_flat = jax.lax.reshape(h_final, (flat_batch, seq_len, cfg.hidden_dim), out_sharding=flat_hidden_spec)
518 
519        # Score the whole global batch (all microbatches) in ONE fused-CE call with a
520        # single mean reduction -- identical to the non-pipelined oracle, which means
521        # over all batch*seq tokens at once (not a mean of per-microbatch means).
522        def _head_mean(fn, fgn, op, h):
523            return _head_loss(fn, fgn, op, h, labels, weights_flat)
524 
525        ce_loss, head_vjp = jax.vjp(_head_mean, final_norm, final_gated_norm, output_proj, h_final_flat)
526        g_final_norm, g_final_gated, g_output_proj, dh_final = head_vjp(jnp.ones((), jnp.float32))
527 
528        z_total = jax.lax.psum(z_total, STAGE_AXIS)
529        # Each microbatch's per-layer z-loss is a token mean over its own rows; the
530        # full-batch mean (what the oracle computes) is the mean over microbatches of
531        # those, so divide the per-(stage,microbatch) sum by M as well as num_layers.
532        aux = cfg.router_z_loss_coef * (z_total / num_layers / num_microbatches)
533        total_loss = ce_loss + aux
534 
535        # --- Backward sweep: reverse-time B pass. dy seeds from dh_final on the last
536        # stage, dbuf (ppermuted upstream) elsewhere; dx ships upstream; embed/stage
537        # grads accumulate. Aux z-loss cotangent is router_z_loss_coef / num_layers
538        # on every valid slot (matching how z_total enters the loss).
539        g_embed_token = jax.tree_util.tree_map(jnp.zeros_like, token_embed)
540        g_embed_norm = jax.tree_util.tree_map(jnp.zeros_like, embed_norm)
541        g_embed_gated = jax.tree_util.tree_map(jnp.zeros_like, embed_gated_norm)
542        g_stage = jax.tree_util.tree_map(jnp.zeros_like, stage_blocks)
543 
544        aux_cot_scale = cfg.router_z_loss_coef / num_layers / num_microbatches
545        microbatch = hidden_shape[0]
546        dbuf = zero_hidden
547        for t in reversed(range(T)):
548            valid = valid_by_t[t]
