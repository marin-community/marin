5 
6This is a deliberately legible re-implementation of the grug MoE architecture
7(``experiments/grug/moe/model.py``) with every mesh-coupled construct removed:
8no ``jax.sharding.reshard``, no ``shard_map``, no abstract-mesh lookups. The
9production model's routing path uses ``reshard`` calls whose axes turn ``Manual``
10inside the pipeline's ``shard_map``; this module stays pure SPMD-identical JAX so
11it can be driven by the zero-bubble pipeline primitive.
12

14

16  no sharded beta, no dynamic router bias).
17- Experts are replicated and dispatched by computing *all* experts and taking a
18  gate-weighted sum over the selected ones — exact, no capacity dropping.
19- SwiGLU expert MLPs (silu(gate) * up, then down).
20- Multi-head attention with rotary embeddings and a manual causal softmax; GQA is
21  supported via key/value head expansion.
22

10stage owns a contiguous slice of the model's layers and the microbatches stream
11through the stages.

14

129        valid = (m >= 0) & (m < M)
130        m_clip = jnp.clip(m, 0, M - 1)

132        tok_m = jax.lax.dynamic_index_in_dim(tokens, m_clip, axis=0, keepdims=False)
133        embedded = model.embed_fn(params.embed, tok_m)
134        stage_in = jnp.where(is_first, embedded, buf)
135 
136        stage_out, stage_aux = model.stage_fn(stage_params, stage_in)
137        # Per-stage aux (e.g. router z-loss) is local to a stage's own layers and
138        # summed across stages — count it on every valid slot, not just the last.
139        total_loss = total_loss + jnp.where(valid, stage_aux, 0.0)
140 
141        tgt_m = jax.lax.dynamic_index_in_dim(targets, m_clip, axis=0, keepdims=False)
142        loss_m = model.head_loss_fn(params.head, stage_out, tgt_m)
143        total_loss = total_loss + jnp.where(is_last & valid, loss_m, 0.0)
144 
145        buf = jax.lax.ppermute(stage_out, STAGE_AXIS, fwd_perm)
146 
147    # Sum loss+aux across stages (head loss on last stage, aux on all), mean over microbatches.
148    total_loss = jax.lax.psum(total_loss, STAGE_AXIS) / M

205def _zero_bubble_body(
206    params: PipelineParams,
207    tokens: jax.Array,
208    targets: jax.Array,
209    *,
210    model: PipelineModel,
211    num_stages: int,
212    num_microbatches: int,
213    hidden_shape: tuple[int, ...],
214    hidden_dtype: jnp.dtype,
215) -> tuple[jax.Array, PipelineParams]:
216    """Explicit forward + split-backward pipeline, run inside a shard_map.
217 
218    The backward is split into two passes that the schedule decouples:
219 
220    - **B pass** (reverse time): per timestep compute ``dx`` (cotangent w.r.t.
221      the stage input) and ppermute it upstream. This is the critical path that
222      unblocks stage ``s-1``; it also seeds the embed gradient on stage 0.
223    - **W pass**: replay the stored output cotangents through ``vjp_p`` to get the
224      weight gradient ``dp``. W depends on nothing downstream, so it is fully
225      deferred here (and on hardware is slotted into the warmup/cooldown bubbles).
226 
227    B and W use *separate* vjp closures so the deferred W is genuine compute, not
228    a buffered add — that decoupling is what makes the bubble fillable.
229    """
230    sid = jax.lax.axis_index(STAGE_AXIS)
231    S = num_stages
232    M = num_microbatches
233    T = M + S - 1
234    fwd_perm = [(i, i + 1) for i in range(S - 1)]
235    bwd_perm = [(i + 1, i) for i in range(S - 1)]
236    is_first = sid == 0
237    is_last = sid == (S - 1)
238 
239    stage_params = jax.tree_util.tree_map(lambda x: x[0], params.stage)
240 
241    # Forward sweep: keep per-timestep vjp closures + outputs (static-keyed by t).
242    vjp_x_by_t: list[Any] = [None] * T
243    vjp_p_by_t: list[Any] = [None] * T
244    embed_vjp_by_t: list[Any] = [None] * T
245    out_by_t: list[Any] = [None] * T
246    tgt_by_t: list[Any] = [None] * T
247    valid_by_t: list[Any] = [None] * T
248 
249    buf = jnp.zeros(hidden_shape, hidden_dtype)
250    total_loss = jnp.zeros((), jnp.float32)
251    for t in range(T):
252        m = t - sid
253        valid = (m >= 0) & (m < M)
254        m_clip = jnp.clip(m, 0, M - 1)
255        tok_m = jax.lax.dynamic_index_in_dim(tokens, m_clip, axis=0, keepdims=False)
256        tgt_m = jax.lax.dynamic_index_in_dim(targets, m_clip, axis=0, keepdims=False)
257 
258        # Loop vars are bound as lambda defaults: jax.vjp traces each immediately,
259        # so the closures capture this iteration's tensors (B023-safe).
260        embedded, embed_vjp = jax.vjp(lambda e, _t=tok_m: model.embed_fn(e, _t), params.embed)
261        stage_in = jnp.where(is_first, embedded, buf)
262        # stage_fn returns (hidden, aux); vjp closures take a (d_hidden, d_aux) cotangent.
263        (stage_out, stage_aux), vjp_x = jax.vjp(lambda x, _sp=stage_params: model.stage_fn(_sp, x), stage_in)
264        _, vjp_p = jax.vjp(lambda p, _si=stage_in: model.stage_fn(p, _si), stage_params)
265 
266        total_loss = total_loss + jnp.where(valid, stage_aux, 0.0)
267        loss_m = model.head_loss_fn(params.head, stage_out, tgt_m)
268        total_loss = total_loss + jnp.where(is_last & valid, loss_m, 0.0)
269 
270        vjp_x_by_t[t] = vjp_x
271        vjp_p_by_t[t] = vjp_p
272        embed_vjp_by_t[t] = embed_vjp
273        out_by_t[t] = stage_out
274        tgt_by_t[t] = tgt_m
275        valid_by_t[t] = valid
276 
277        buf = jax.lax.ppermute(stage_out, STAGE_AXIS, fwd_perm)
278 
279    total_loss = jax.lax.psum(total_loss, STAGE_AXIS) / M
280 
281    # Backward B pass (reverse time): dx upstream + head/embed grads + store dy.
282    g_embed = jax.tree_util.tree_map(jnp.zeros_like, params.embed)
283    g_head = jax.tree_util.tree_map(jnp.zeros_like, params.head)
284    dy_by_t: list[Any] = [None] * T
285    dbuf = jnp.zeros(hidden_shape, hidden_dtype)
286    for t in reversed(range(T)):
287        valid = valid_by_t[t]
288        # Cotangent on this microbatch's loss is 1/M, but only the last stage's
289        # valid slots actually feed the loss; everything else contributes nothing.
290        last_valid = is_last & valid
291        loss_w = jnp.where(last_valid, 1.0 / M, 0.0)
292        head_vjp = jax.vjp(lambda h, o, _tg=tgt_by_t[t]: model.head_loss_fn(h, o, _tg), params.head, out_by_t[t])[1]
293        g_head_t, dout = head_vjp(loss_w)
294        dy = jnp.where(is_last, dout, dbuf)
295        dy_by_t[t] = dy
296        g_head = _tree_add(g_head, g_head_t)
297 
298        # Aux (stage-local) is part of the loss on every valid slot; cotangent 1/M.
299        aux_cot = jnp.where(valid, 1.0 / M, 0.0)
300        (dx,) = vjp_x_by_t[t]((dy, aux_cot))
301        # embed grad lives only on stage 0's valid slots.
302        first_valid = is_first & valid
303        (g_embed_t,) = embed_vjp_by_t[t](jnp.where(first_valid, dx, jnp.zeros_like(dx)))
304        g_embed = _tree_add(g_embed, g_embed_t)
305        dbuf = jax.lax.ppermute(dx, STAGE_AXIS, bwd_perm)
306 
307    # Backward W pass (deferred): replay stored cotangents through vjp_p.
308    g_stage = jax.tree_util.tree_map(jnp.zeros_like, stage_params)
309    for t in range(T):
310        aux_cot = jnp.where(valid_by_t[t], 1.0 / M, 0.0)
311        (dp,) = vjp_p_by_t[t]((dy_by_t[t], aux_cot))
312        g_stage = _tree_add(g_stage, dp)
313 
314    # embed/head are replicated, so a vjp w.r.t. them inside shard_map already
315    # psums the cotangent across the stage axis: g_embed/g_head come back correct
316    # on every device, no manual psum. stage params are sharded over `stage`, so
317    # each device keeps its own stage grad; re-add the leading shard axis.
318    g_stage = jax.tree_util.tree_map(lambda x: x[None], g_stage)
319    return total_loss, PipelineParams(embed=g_embed, stage=g_stage, head=g_head)
