4"""Delayed-gradient optimizer wrappers for studying pipeline-parallel staleness.
5 
6Pipeline-parallel training with a throughput-optimal async schedule applies a
7gradient that was computed ``tau`` weight-versions ago. We study that regime
8*without building pipeline parallelism* by injecting a controlled gradient delay
9in software: the wrapper keeps a depth-``tau`` FIFO of past gradients (and the
10weights they were computed at) inside the optimizer state, and feeds the inner
11optimizer the *stale* gradient each step. This exactly reproduces constant-delay
12asynchronous SGD.
13 
14The FIFO and any correction statistics live in ``opt_state`` — they are the
15"O(weights) extra optimizer state" we are budgeting for — so the canonical grug
16train loop needs no changes; we only swap the optimizer config. At ``tau == 0``
17the wrapper is a pass-through and is bit-identical to the inner optimizer.
18 
19Correctors (applied to the stale gradient before the inner optimizer sees it):
20 
21- ``none``        — naive async SGD; apply the stale gradient as-is.
22- ``dc_asgd``     — DC-ASGD delay compensation (Zheng et al. 2017):
23                    ``g + lambda * (g (.) g) (.) (w_t - w_stale)`` using the
24                    instantaneous squared stale gradient as the diagonal-Hessian
25                    proxy.
26- ``dc_asgd_ema`` — the same correction but with the diagonal curvature read from
27                    an EMA of the squared gradient (i.e. Adam/RMSProp's second
28                    moment ``v_t``). This tests the "reuse the preconditioner
29                    state as the curvature term, near-free" hypothesis.
30 
31For Muon the corrected gradient is fed *before* Newton-Schulz orthogonalization,
32so the correction acts on the momentum/direction and orthogonalization then
33renormalizes the magnitude.
34 
35``weight_pred`` is different in kind: it is a *forward-side* corrector, not a
36gradient correction. Instead of patching the stale gradient, it asks the train
37step to evaluate the gradient at *predicted* weights ``W_hat = w - tau*lr*dW``
38(extrapolating the most recent applied update ``dW`` forward by ``tau`` steps) so
39that, once the gradient is delayed and applied, it lands on the weights it was
40meant for. For Muon ``dW`` is the orthogonalized update (post-orthogonalization
41prediction). The optimizer cannot do this alone — it exposes the predicted offset
42via :meth:`make_forward_predictor` and the grug train step computes the forward
43there; see ``experiments/grug/moe/train.py``.

248def _make_train_step(
249    optimizer: optax.GradientTransformation,
250    mp: jmp.Policy,
251    *,
252    z_loss_weight: float,
253    ema_beta: float | None,
254    watch_config: WatchConfig | None = None,
255):
256    one = jnp.array(1, dtype=jnp.int32)
257    z_loss = z_loss_weight if z_loss_weight > 0 else None
258    if watch_config is not None:
259        if isinstance(watch_config.watch_targets, str):
260            watch_targets = tuple(t.strip() for t in watch_config.watch_targets.split(","))
261        else:
262            watch_targets = tuple(watch_config.watch_targets)
263    else:
264        watch_targets = ()
265 
266    @functools.partial(jax.jit, donate_argnums=(0,), static_argnames=("compute_watch",))
267    def train_step(state: GrugTrainState, batch, *, compute_watch: bool = False):
268        def loss_fn(params):
269            compute_params = mp.cast_to_compute(params)
270            return compute_params.next_token_loss(
271                batch.tokens,
272                batch.loss_weight,
273                mask=batch.attn_mask,
274                reduction="mean",
275                logsumexp_weight=z_loss,
276            )
277 
278        loss, grads = jax.value_and_grad(loss_fn)(state.params)
279        updates, opt_state = optimizer.update(grads, state.opt_state, state.params)
280        params = optax.apply_updates(state.params, updates)
281 
282        if ema_beta is None:
283            ema_params = None
284        else:
285            if state.ema_params is None:
286                raise ValueError("ema_params must be initialized when ema_beta is set.")
287            ema_params = jax.tree_util.tree_map(
288                lambda old, new: ema_beta * old + (1.0 - ema_beta) * new,
289                state.ema_params,
290                params,
291            )
292 
293        watch_stats = None
294        if watch_config is not None and compute_watch:
295            watch_stats = compute_watch_stats(
296                watch_targets=watch_targets,
297                include_norms=watch_config.include_norms,
298                include_per_parameter_norms=watch_config.include_per_parameter_norms,
299                include_histogram=watch_config.include_histograms,
300                split_scan_layers=watch_config.split_scan_layers,
301                params=state.params,
302                grads=grads,
303                updates=updates,
304                opt_state=state.opt_state,
305                model_tree_type=type(state.params),
306            )
307 
308        next_state = dataclasses.replace(
309            state,
310            step=state.step + one,
311            params=params,
312            opt_state=opt_state,
313            ema_params=ema_params,
314        )
315 
316        return next_state, {"train/loss": loss}, watch_stats
317 
318    return train_step

322    """Run grug training locally inside a Fray worker."""
323    trainer = config.trainer.trainer
324    trainer.initialize()
325    levanter.tracker.log_configuration(config)
326 
327    run_id = trainer.id
328    if run_id is None:
329        raise ValueError("trainer.id was not initialized")
330 
331    optimizer = config.optimizer.build(trainer.num_train_steps)
332    watch_config = trainer.watch
333    train_step = _make_train_step(
334        optimizer,
335        trainer.mp,
336        z_loss_weight=config.trainer.z_loss_weight,

339    )

341    data_key, model_key = jax.random.split(jax.random.PRNGKey(trainer.seed), 2)
342    if config.trainer.data_seed is not None:
343        data_key = jax.random.PRNGKey(config.trainer.data_seed)
344 
345    # Grug uses raw PartitionSpecs rather than Trainer's logical axis mapping.

351        train_dataset = build_train_dataset(
352            config.data,
353            max_seq_len=config.model.max_seq_len,
354            batch_schedule=batch_schedule,
355            key=data_key,
356        )
357        train_loader = build_train_loader(
358            train_dataset,
359            batch_schedule=batch_schedule,
360            mesh=mesh,
361            batch_pspec=config.trainer.train_batch_pspec,
362        )
363 
364        @jax.jit
365        def _init_state(model_rng):
366            return initial_state(
367                config.model,
368                optimizer=optimizer,
369                mp=trainer.mp,
370                key=model_rng,
371                ema_beta=config.trainer.ema_beta,
372            )
373 
374        state = _init_state(model_key)
375 
376        checkpointer = trainer.checkpointer.create(run_id)
377        state = restore_grug_state_from_checkpoint(
378            state,
379            checkpoint_search_paths=trainer.checkpoint_search_paths(run_id),

387        flops_per_example, flops_summary = _compute_flops(model_config=config.model)

391        evaluator = None
392        if eval_cfg is not None:
393            evaluator = build_tagged_evaluator(

396                mesh=mesh,
397                eval_cfg=eval_cfg,

402        profiler_enabled = profiler_cfg.is_enabled and profiler_num_steps > 0

404        log_every = max(1, config.trainer.log_every)
405        iterator = LoadingTimeTrackerIterator(train_loader.iter_from_step(int(state.step)))
406 
407        state_callbacks = StateCallbackRunner[GrugTrainState](
408            step_getter=lambda s: s.step,
409            model_getter=lambda s: s.params,
410            eval_model_getter=lambda s: s.ema_params if s.ema_params is not None else s.params,
411            opt_state_getter=lambda s: s.opt_state,
412        )
413        state_callbacks.add_hook(
414            callbacks.log_performance_stats(config.model.max_seq_len, batch_schedule, flops_per_example),
415            every=log_every,
416        )

418        state_callbacks.add_hook(callbacks.log_step_info(trainer.num_train_steps), every=log_every)
419        if profiler_enabled:
420            state_callbacks.add_hook(
421                callbacks.profile(
422                    str(trainer.log_dir / run_id / "profiler"),
423                    profiler_cfg.start_step,
424                    profiler_num_steps,
425                    profiler_cfg.perfetto_link,
426                ),
427                every=1,
428            )
429        state_callbacks.add_hook(_make_mixture_stage_callback(train_dataset, batch_schedule), every=1)
430        if evaluator is not None and eval_cfg is not None:
431            interval = eval_cfg.steps_per_eval
432            eval_ema = eval_cfg.eval_ema and config.trainer.ema_beta is not None
433            if interval is not None and interval > 0 and (eval_cfg.eval_current or eval_ema):
434                state_callbacks.add_hook(
435                    cb_tagged_evaluate(
436                        evaluator,
437                        prefix=eval_cfg.prefix,
438                        eval_current=eval_cfg.eval_current,
439                        eval_ema=eval_ema,
440                    ),
441                    every=interval,
442                )
443 
444        last_loss: float | jax.Array = 0.0
445        last_step_duration = 0.0
446 
447        # Main optimization loop.
448        try:
449            while int(state.step) < trainer.num_train_steps:
450                with jax.profiler.TraceAnnotation("load_batch"):
451                    batch = next(iterator)
452                step_start = time.perf_counter()
453                current_step = int(state.step)
454                # grad_watch runs only on its configured interval.
455                compute_watch = (
456                    watch_config.is_enabled and watch_config.interval > 0 and current_step % watch_config.interval == 0
457                )
458                state, metrics, watch_stats = train_step(state, batch, compute_watch=compute_watch)
459                step = int(state.step) - 1
460 
461                jax.block_until_ready(metrics["train/loss"])
462                duration = time.perf_counter() - step_start
463                hook_start = time.perf_counter()
464                with jax.profiler.TraceAnnotation("callbacks"):
465                    state_callbacks.run(state, loss=metrics["train/loss"], step_duration=duration)
466                    last_loss = metrics["train/loss"]
467                    last_step_duration = duration
468                    levanter.tracker.log({"throughput/hook_time": time.perf_counter() - hook_start}, step=step)
469                    levanter.tracker.log({"throughput/loading_time": iterator.this_load_time}, step=step)
470 
471                    if watch_stats is not None:
472                        levanter.tracker.log(watch_stats, step=step)
473 
474                if checkpointer is not None:
475                    checkpointer.on_step(tree=state, step=int(state.step))
476        except BaseException:
477            logger.exception(
478                "Fatal error in grug training loop; skipping final callbacks/checkpoint to preserve root cause"
479            )
480            raise
481        else:
482            # Mirror classic trainer behavior: force callbacks on the last completed step.
483            state_callbacks.run(state, loss=last_loss, step_duration=last_step_duration, force=True)
484            if checkpointer is not None:
485                checkpointer.on_step(tree=state, step=int(state.step), force=True)
486                checkpointer.wait_until_finished()
487 
488    levanter.tracker.current_tracker().finish()
